--PUPRPOSE:  THIS SCRIPT PULLS STUDENTS BY PROMPTED HIGH SCHOOL CODE.  


SELECT DISTINCT SPRIDEN_ID AS "A NUMBER",
       SPRIDEN_LAST_NAME AS "LAST",
        SPRIDEN_fIRST_NAME AS "FIRST"
  
       FROM SPRIDEN, SORHSCH
       
 WHERE SPRIDEN_PIDM = SORHSCH_PIDM
 
 AND      SORHSCH.SORHSCH_SBGI_CODE = :HIGH_SCHOOL_CODE
 
 
 AND     spriden_change_ind is NULL
       
       ORDER BY 2;
